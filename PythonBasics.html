Code Example: Point and Rectangle Classes

Below is the Python code example:

from __future__ import print_function, division

class Point:
    def __init__(self, x=0, y=0):
        """Initializes a point with x and y attributes."""
        self.x = x
        self.y = y

    def print_point(self):
        """Print a Point object in human-readable format."""
        print('(%g, %g)' % (self.x, self.y))


class Rectangle:
    """Represents a rectangle."""
    def __init__(self, width=0, height=0, corner=Point(0,0)):
        self.width = width
        self.height = height
        self.corner = corner

    def area(self):
        """Calculates the area of the rectangle."""
        return self.width * self.height

    def find_center(self):
        """Returns a Point at the center of a Rectangle."""
        p = Point()
        p.x = self.corner.x + self.width / 2.0
        p.y = self.corner.y + self.height / 2.0
        return p

    def grow_rectangle(self, dwidth, dheight):
        """Modifies the Rectangle by adding to its width and height."""
        self.width += dwidth
        self.height += dheight


def main():
    canvas = Point(4, 5)
    print('canvas', end=' ')
    canvas.print_point()

    rect = Rectangle(100, 200, Point(0, 0))

    center = rect.find_center()
    print('center', end=' ')
    center.print_point()

    print('width', end=' ')
    print(rect.width)
    print('height', end=' ')
    print(rect.height)
    print('area', end=' ')
    print(rect.area())

    print('grow')
    rect.grow_rectangle(50, 100)

    print('new width', end=' ')
    print(rect.width)
    print('new height', end=' ')
    print(rect.height)
    print('new area', end=' ')
    print(rect.area())


if __name__ == '__main__':
    main()
